function nombre (props){
const[state,setState]=UseState({
titulo:"",

cuenta:1
});

const disminuir =()=>{
 setState({
	if (state.cuenta <= 0){
	setState({
	...state,cuenta:0
	});
			}else{setState({...state,cuenta:state.cuenta - 1}); }
	});
}

const aumentar =()=>{
 setState({
	...state,cuenta:state.cuenta + 1
	});
}

return(
<p>{state.titulo}</p>
<p>{state.subtitulo}</p>
<button onClick={disminuir}>-</button>
<button onClick={aumentar}>+</button>

);
}
